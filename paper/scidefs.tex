\newcount\posveccount
\newcommand*\posvec[1]{
        \global\posveccount#1
        [
        \posvecnext
}
\def\posvecnext#1{
        #1
        \global\advance\posveccount-1
        \ifnum\posveccount>0
                ,\
                \expandafter\posvecnext
        \else
                ]^{\mathrm{T}}
        \fi
}
